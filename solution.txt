function getAllConsecutiveSubstrings(str) {
    let substrings = [];
    
    for (let i = 0; i < str.length; i++) {
        for (let j = i + 1; j <= str.length; j++) {
            substrings.push(str.slice(i, j));
        }
    }
    
    return substrings;
}

console.log("Testing with 'abc':");
let result1 = getAllConsecutiveSubstrings('abc');
console.log(result1);

console.log("\nTesting with 'a':");
let result2 = getAllConsecutiveSubstrings('a');
console.log(result2);

console.log("\nTesting with empty string '':");
let result3 = getAllConsecutiveSubstrings('');
console.log(result3);
